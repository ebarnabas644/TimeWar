295  protected virtual void Movement()
296  {
297  Point newPoint = this.Move();
298  if (Math.Abs(this.Character.MovementVector.X) < this.MaxMovementSpeed)
299  {
300  this.AddToVector(newPoint.X, 0);
301  }
302 
303  this.AddToVector(0, newPoint.Y);
304 
305  if (this.GroundCollision(new Point(0, newPoint.Y)))
306  {
307  if (this.CommandManager.IsFinished)
308  {
309  while (this.GroundCollision(new Point(0, newPoint.Y)))
310  {
311  this.Character.Position = new Point(this.Character.Position.X, this.Character.Position.Y - 1);
312  }
313  }
314  }
315 
316  if (!this.GroundCollision(new Point(0, this.Acceleration)))
317  {
318  if (this.CommandManager.IsFinished)
319  {
320  this.AddToVector(0, this.Acceleration);
321 
322  if ((!this.GroundCollision(new Point(0, this.Acceleration + 1)) && this.Acceleration < GravityAcceleration) && !this.AccelerationStopwatch.IsRunning)
323  {
324  this.AccelerationStopwatch.Start();
325  }
326 
327  if (this.AccelerationStopwatch.ElapsedMilliseconds > 100)
328  {
329  this.Acceleration++;
330  this.AccelerationStopwatch.Restart();
331  }
332  }
333  }
334 
335  if (!this.WallCollision(newPoint) && !this.WallCollision(newPoint, false))
336  {
337  this.Character.Position = new Point(this.Character.Position.X + this.Character.MovementVector.X, this.Character.Position.Y);
338  }
339 
340  if (!this.GroundCollision(newPoint) && (!this.WallCollision(newPoint) && !this.WallCollision(newPoint, false)))
341  {
342  if (!this.TopCollision(newPoint))
343  {
344  this.Character.Position = new Point(this.Character.Position.X, this.Character.Position.Y + this.Character.MovementVector.Y);
345  }
346  }
347 
348  MoveCommand moveCommand = new MoveCommand(this.Character, this.Character.Position, this.Model);
349  this.CommandManager.AddCommand(moveCommand);
350 
351  if (this.Character.MovementVector.X > 0)
352  {
353  this.AddToVector(-1, 0);
354  }
355 
356  if (this.Character.MovementVector.X < 0)
357  {
358  this.AddToVector(1, 0);
359  }
360  }
