301  protected virtual void Movement()
302  {
303  Point newPoint = this.Move();
304  if (Math.Abs(this.Character.MovementVector.X) < this.MaxMovementSpeed)
305  {
306  this.AddToVector(newPoint.X, 0);
307  }
308 
309  this.AddToVector(0, newPoint.Y);
310 
311  if (this.GroundCollision(new Point(0, newPoint.Y)))
312  {
313  if (this.CommandManager.IsFinished)
314  {
315  while (this.GroundCollision(new Point(0, newPoint.Y)))
316  {
317  this.Character.Position = new Point(this.Character.Position.X, this.Character.Position.Y - 1);
318  }
319  }
320  }
321 
322  if (!this.GroundCollision(new Point(0, this.Acceleration)))
323  {
324  if (this.CommandManager.IsFinished)
325  {
326  this.AddToVector(0, this.Acceleration);
327 
328  if ((!this.GroundCollision(new Point(0, this.Acceleration + 1)) && this.Acceleration < GravityAcceleration) && !this.AccelerationStopwatch.IsRunning)
329  {
330  this.AccelerationStopwatch.Start();
331  }
332 
333  if (this.AccelerationStopwatch.ElapsedMilliseconds > 100)
334  {
335  if (Math.Abs(this.Acceleration) <= 2)
336  {
337  this.Acceleration++;
338  }
339 
340  this.AccelerationStopwatch.Restart();
341  }
342  }
343  }
344 
345  if (!this.WallCollision(newPoint) && !this.WallCollision(newPoint, false))
346  {
347  this.Character.Position = new Point(this.Character.Position.X + this.Character.MovementVector.X, this.Character.Position.Y);
348  }
349 
350  if (!this.GroundCollision(newPoint) && (!this.WallCollision(newPoint) && !this.WallCollision(newPoint, false)))
351  {
352  if (!this.TopCollision(newPoint))
353  {
354  this.Character.Position = new Point(this.Character.Position.X, this.Character.Position.Y + this.Character.MovementVector.Y);
355  }
356  }
357 
358  MoveCommand moveCommand = new MoveCommand(this.Character, this.Character.Position, this.Model);
359  this.CommandManager.AddCommand(moveCommand);
360  if (this.Character.MovementVector.X > 0)
361  {
362  this.AddToVector(-1, 0);
363  }
364 
365  if (this.Character.MovementVector.X < 0)
366  {
367  this.AddToVector(1, 0);
368  }
369  }
