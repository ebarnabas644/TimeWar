117  protected override Point Move()
118  {
119  int x = 0;
120  int y = 0;
121  if (!this.IsPlayerDetected)
122  {
123  if (this.movementDirStopwatch.ElapsedMilliseconds > this.movementDirTime)
124  {
125  this.moveDir = RandomNumberGenerator.GetInt32(3);
126  this.movementDirTime = RandomNumberGenerator.GetInt32(this.MaxMoveTime);
127  this.movementDirStopwatch.Restart();
128  }
129 
130  switch (this.moveDir)
131  {
132  case 0:
133  x -= 2;
134  break;
135 
136  case 1:
137  x += 2;
138  break;
139 
140  case 2:
141  x += 0;
142  break;
143 
144  default:
145  x += 0;
146  break;
147  }
148  }
149  else
150  {
151  if (this.playerDetectionStopwatch.ElapsedMilliseconds < this.DetectionTime)
152  {
153  int distance = this.PlayerDistance();
154  if (!this.isPlayerVisible)
155  {
156  this.followDistance = 0;
157  }
158  else
159  {
160  this.followDistance = this.DefaultFollowDistance;
161  }
162 
163  if (this.PixelToTile(this.Character.Position.X) > this.LastKnownPlayerLocation.X && distance >= this.followDistance)
164  {
165  x -= 2;
166  }
167 
168  if (this.PixelToTile(this.Character.Position.X) < this.LastKnownPlayerLocation.X && distance >= this.followDistance)
169  {
170  x += 2;
171  }
172 
173  if (this.PixelToTile(this.Character.Position.Y) < this.LastKnownPlayerLocation.Y)
174  {
175  y += this.Jump();
176  }
177  }
178  else
179  {
180  this.moveDir = RandomNumberGenerator.GetInt32(4);
181  this.movementDirStopwatch.Restart();
182  }
183  }
184 
185  int rnd = RandomNumberGenerator.GetInt32(10000);
186  if (rnd < 100)
187  {
188  y += this.Jump();
189  }
190 
191  return new Point(x, y);
192  }

274  int dir = -1;

277  if (right)
278  {
279  dir = 1;
280  startPoint.X++;
281  }
282 
283  for (int i = 0; i < this.DetectionRange; i++)
284  {
285  height++;

288  {
289  return false;
