123  private void Movement(Bullet bullet)
124  {

134  AcceleratingMovement(bullet);

138  break;
139  default:

144  bullet.Position = new Point(bullet.Position.X + (int)bullet.MoveVector.X, bullet.Position.Y + (int)bullet.MoveVector.Y);
145  MoveCommand cmd = new MoveCommand(bullet, bullet.Position, this.model);
