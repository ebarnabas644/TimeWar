156  private void Win_MouseDown(object sender, System.Windows.Input.MouseEventArgs e)
157  {
158  if (e.LeftButton == MouseButtonState.Pressed)
159  {
160  System.Windows.Point clickPosition = e.GetPosition(this);
161  System.Drawing.Point clickRelativePos = new System.Drawing.Point(this.model.Camera.GetRelativeCharacterPosX, this.model.Camera.GetRelativeCharacterPosY);
162  clickRelativePos.X -= (int)clickPosition.X;
163  clickRelativePos.X *= -1;
164  clickRelativePos.Y -= (int)clickPosition.Y;
165  clickRelativePos.X += this.model.Hero.Position.X;
166  clickRelativePos.Y -= this.model.Hero.Position.Y;
167  clickRelativePos.Y *= -1;
168  this.model.Hero.CanAttack = true;
169  this.model.Hero.ClickLocation = clickRelativePos;
170  }
