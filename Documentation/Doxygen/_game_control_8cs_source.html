122  public void SaveEndGame()
123  {
124  var profile = this.factory.ViewerLogic.GetSelectedProfile();
125  if (profile != null)
126  {
127  var q = this.factory.ViewerLogic.GetMaps().Where(x => x.MapName == this.MapName && x.PlayerId == profile.PlayerId).SingleOrDefault();
128  if (q != null)
129  {
130  this.factory.ManagerLogic.ModifyMap(new MapRecord { MapName = this.MapName, RunTime = this.gm.EndTime, MapRecordId = q.MapRecordId, PlayerId = q.PlayerId });
131  }
132  else
133  {
134  this.factory.ManagerLogic.CreateMap(new MapRecord { MapName = this.MapName, RunTime = this.gm.EndTime, PlayerId = profile.PlayerId });
135  }
136 
137  PlayerProfile modifiedPlayer = new PlayerProfile();
138  modifiedPlayer.PlayerId = profile.PlayerId;
139  modifiedPlayer.Selected = true;
140  modifiedPlayer.CompletedRuns = ++profile.CompletedRuns;
141  modifiedPlayer.TotalDeaths = profile.TotalDeaths + this.gm.EndDeaths;
142  modifiedPlayer.TotalKills = profile.TotalKills + this.gm.EndKills;
143  if (profile.Save != null)
144  {
145  this.factory.ManagerLogic.DeleteSave(profile.Save);
146  }
147 
148  this.factory.ManagerLogic.ModifyProfile(modifiedPlayer);
149  }
150  }

288  private void Tick()
289  {
290  if (!this.Exit)
291  {
292  if (!this.IsPaused)
293  {
294  if (this.model.CurrentWorld.EnemiesLoaded)
295  {
296  this.model.CurrentWorld.EnemiesLoaded = false;
297  Stopwatch st = new Stopwatch();
298  this.characterLogic.AttackTime = 200;
299  this.characterLogic.MaxJumpHeight = this.characterLogic.DefaultJumpHeight;
300  this.characterLogic.Character.IsInvincible = false;
301  this.enemyLogic.GetEnemies();
302  this.pointOfInterestLogics.GetPOIs();
303  }
304 
305  this.characterLogic.OneTick();
306  this.enemyLogic.TickEnemies();
307  this.pointOfInterestLogics.TickPois();
308  this.bulletLogic.Addbullets((ICollection<Bullet>)this.model.CurrentWorld.GetBullets);
309  this.bulletLogic.OneTick();
310  }
311 
312  if (this.model.LevelFinished)
313  {
314  lock (this.factory)
315  {
316  this.gm.EndKills = this.model.Hero.Kills;
317  this.gm.EndDeaths = this.model.Hero.Deaths;
318  this.gm.EndTime = this.time.Elapsed;
319  this.gm.EndVisibility = true;
320  this.LevelFinished = true;
321  this.IsPaused = true;
322  this.time.Stop();
323  this.deltatime.Stop();
324  }
325  }
326  }
327  else
328  {
329  this.gm.EndVisibility = false;
330  this.win.KeyDown -= this.Win_KeyDown;
331  this.win.KeyUp -= this.Win_KeyUp;
332  this.win.SizeChanged -= this.Win_SizeChanged;
333  this.win.MouseMove -= this.Win_MouseMove;
334  this.win.MouseDown -= this.Win_MouseDown;
335  this.win.MouseWheel -= this.Win_MouseScroll;
336  this.timer.Dispose();
337  this.Dispose();
338  CompositionTarget.Rendering -= (sender, args) => this.InvalidateVisual();
339  }
340  }

342  private void Win_MouseScroll(object sender, MouseWheelEventArgs e)
343  {
344  if (e.Delta > 0)
345  {
346  this.mouseScrollPos++;
347  }
348  else if (e.Delta < 0)
349  {
350  this.mouseScrollPos--;
351  }
352 
353  switch (this.mouseScrollPos % this.model.Hero.NumOfWeaponUnlocked)
354  {
355  case 0:
356  this.model.Hero.TypeOfBullet = BulletType.Basic;
357  break;
358 
359  case 1:
360  this.model.Hero.TypeOfBullet = BulletType.Accelerating;
361  break;
362 
363  case 2:
364  this.model.Hero.TypeOfBullet = BulletType.Bouncing;
365  break;
366 
367  case 3:
368  this.model.Hero.TypeOfBullet = BulletType.CurvedBouncing;
369  break;
370 
371  default:
372  this.model.Hero.TypeOfBullet = BulletType.Basic;
373  break;
374  }
375  }

383  private void Win_MouseDown(object sender, System.Windows.Input.MouseEventArgs e)
384  {
385  if (e.LeftButton == MouseButtonState.Pressed && !this.IsPaused)
386  {
387  System.Windows.Point clickPosition = e.GetPosition(this);
388  System.Drawing.Point clickRelativePos = new System.Drawing.Point(this.model.Camera.GetRelativeCharacterPosX, this.model.Camera.GetRelativeCharacterPosY);
389  clickRelativePos.X -= (int)clickPosition.X;
390  clickRelativePos.X *= -1;
391  clickRelativePos.Y -= (int)clickPosition.Y;
392  clickRelativePos.X += this.model.Hero.Position.X;
393  clickRelativePos.Y -= this.model.Hero.Position.Y;
394  clickRelativePos.Y *= -1;
395  this.model.Hero.CanAttack = true;
396  this.model.Hero.ClickLocation = clickRelativePos;
397  }
