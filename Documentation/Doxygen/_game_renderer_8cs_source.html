115  private static void StateMachine(IGameObject obj)
116  {
117  if (obj.MovementVector.X > 0 && obj.MovementVector.Y > 0)
118  {
119  obj.Stance = Stances.JumpRight;
120  }
121  else if (obj.MovementVector.X < 0 && obj.MovementVector.Y < 0)
122  {
123  obj.Stance = Stances.JumpLeft;
124  }
125  else if (obj.MovementVector.X > 0)
126  {
127  obj.Stance = Stances.Right;
128  }
129  else if (obj.MovementVector.X < 0)
130  {
131  obj.Stance = Stances.Left;
132  }
133  else if (obj.MovementVector.Y > 0 && obj.MovementVector.Y < 0)
134  {
135  if (obj.Stance == Stances.StandRight)
136  {
137  obj.Stance = Stances.JumpRight;
138  }
139  else
140  {
141  obj.Stance = Stances.JumpLeft;
142  }
143  }
144  else if (obj.MovementVector.X == 0 && obj.MovementVector.Y == 0)
145  {
146  if (obj.Stance == Stances.Right || obj.Stance == Stances.JumpRight)
147  {
148  obj.Stance = Stances.StandRight;
149  }
150  else if (obj.Stance == Stances.Left || obj.Stance == Stances.JumpLeft)
151  {
152  obj.Stance = Stances.StandLeft;
153  }
154  }
155  }

180  private Brush GetSpriteBrush(IGameObject obj, bool tiled = false, double boundx = 0, double boundy = 0)
181  {
182  if (!this.spriteBrushes.ContainsKey(obj))
183  {
184  ImageBrush[][] imageBrushes = Sprite.CreateSprite(obj.Height, obj.Width, obj.SpriteFile);
185  if (tiled)
186  {
187  for (int i = 0; i < imageBrushes.Length; i++)
188  {
189  for (int y = 0; y < imageBrushes[i].Length; y++)
190  {
191  imageBrushes[i][y].TileMode = TileMode.Tile;
192  imageBrushes[i][y].Viewport = new Rect(0, 0, (obj.Width * this.model.CurrentWorld.Magnify) / boundx, (obj.Height * this.model.CurrentWorld.Magnify) / boundy);
193  imageBrushes[i][y].ViewportUnits = BrushMappingMode.RelativeToBoundingBox;
194  imageBrushes[i][y].Stretch = Stretch.Fill;
195  }
196  }
197  }
198 
199  this.spriteBrushes.Add(obj, imageBrushes);
200  }
201 
202  if (!obj.StanceLess)
203  {
204  StateMachine(obj);
205  return this.spriteBrushes[obj][(int)obj.Stance][this.currentSprite % this.spriteBrushes[obj][(int)obj.Stance].Length];
206  }
207 
208  return this.spriteBrushes[obj][0][this.currentSprite % this.spriteBrushes[obj][0].Length];
209  }
