52  public void LoadSave(GameModel gameModel)
53  {
54  if (gameModel != null)
55  {
56  PlayerProfile profile = this.viewerLogic.GetSelectedProfile();
57  Save save = this.viewerLogic.GetSave(profile.Save.Id);
58  string enemyStrings = save.Enemydata;
59  string playerString = save.Playerdata;
60 
61  Player character = this.GetPlayer();
62  string[] characterData = playerString.Split(';');
63  character.Position = new Point(Convert.ToInt32(characterData[0], System.Globalization.CultureInfo.CurrentCulture), Convert.ToInt32(characterData[1], System.Globalization.CultureInfo.CurrentCulture));
64  character.CurrentHealth = Convert.ToInt32(characterData[2], System.Globalization.CultureInfo.CurrentCulture);
65  character.CurrentShield = Convert.ToInt32(characterData[3], System.Globalization.CultureInfo.CurrentCulture);
66  character.TypeOfBullet = (BulletType)Enum.Parse(typeof(EnemyType), characterData[4]);
67  character.NumOfWeaponUnlocked = Convert.ToInt32(characterData[5], System.Globalization.CultureInfo.CurrentCulture);
68  this.model.Hero = character;
69 
70  string[] enemies = enemyStrings.Split('!');
71  foreach (string enemy in enemies)
72  {
73  if (!string.IsNullOrEmpty(enemy))
74  {
75  string[] data = enemy.Split(';');
76  Enemy e = new Enemy(new Point(Convert.ToInt32(data[1], System.Globalization.CultureInfo.CurrentCulture), Convert.ToInt32(data[2], System.Globalization.CultureInfo.CurrentCulture)), Convert.ToInt32(data[3], System.Globalization.CultureInfo.CurrentCulture), EnemyInitLogic.BasicEnemyHeight, EnemyInitLogic.BasicEnemyWidth, (EnemyType)Enum.Parse(typeof(EnemyType), data[0]), data[5]);
77  gameModel.CurrentWorld.AddEnemy(e);
78  }
79  }
80  }

137  case 496: gameWorld.AddPOI(new PointOfInterest(POIType.HealthKit, 8, 8, "495", new Point(x, y))); break;
138  case 497: gameWorld.AddPOI(new PointOfInterest(POIType.HighJump, 8, 8, "496", new Point(x, y))); break;
139  case 498: gameWorld.AddPOI(new PointOfInterest(POIType.Invincibility, 8, 8, "497", new Point(x, y))); break;
140  case 499: gameWorld.AddPOI(new PointOfInterest(POIType.RapidFire, 8, 8, "498", new Point(x, y))); break;
141  case 500: gameWorld.AddPOI(new PointOfInterest(POIType.UnlockWeapon, 8, 8, "499", new Point(x, y))); break;
142  case 501: gameWorld.AddPOI(new PointOfInterest(POIType.EnviromentalDamage, 8, 8, "empty", new Point(x, y))); break;

191  private void BuildModel(string mapName)
192  {
193  XDocument mapData = XDocument.Load("Leveldata\\" + mapName + ".tmx");
194  string[][] grounddata = GetGround(mapData);
195  string[][] poidata = GetPoi(mapData);
196  string[][] decodata = GetDecorations(mapData);
197 
198  int gameWorldHeight = grounddata.Length;
199  int gameWorldWidth = grounddata[0].Length;
200  int tileSize = GetTileSize(mapData);
201  this.model.CurrentWorld = new GameWorld(gameWorldHeight, gameWorldWidth, tileSize);
202  this.model.CurrentWorld.WorldName = mapName;
203  FillGround(this.model.CurrentWorld, grounddata);
204  FillPoi(this.model.CurrentWorld, poidata);
205  FillDeco(this.model.CurrentWorld, decodata);
206  if (!this.GameContinued)
207  {
208  string[][] enemydata = GetEnemies(mapData);
209  this.FillEnemy(this.model.CurrentWorld, enemydata);
210  this.model.Hero = this.GetPlayer();
211  }
212  else
213  {
214  this.LoadSave(this.model);
215  }
216  }
