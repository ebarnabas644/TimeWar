142  private void BuildModel(string mapName)
143  {
144  XDocument mapData = XDocument.Load("Leveldata\\" + mapName + ".tmx");
145  string[][] grounddata = GetGround(mapData);
146  string[][] poidata = GetPoi(mapData);
147  string[][] decodata = GetDecorations(mapData);
148  string[][] enemydata = GetEnemies(mapData);
149  int gameWorldHeight = grounddata.Length;
150  int gameWorldWidth = grounddata[0].Length;
151  int tileSize = GetTileSize(mapData);
152  this.model.CurrentWorld = new GameWorld(gameWorldHeight, gameWorldWidth, tileSize);
153  this.model.CurrentWorld.WorldName = mapName;
154  FillGround(this.model.CurrentWorld, grounddata);
155  FillPoi(this.model.CurrentWorld, poidata);
156  FillDeco(this.model.CurrentWorld, decodata);
157  this.FillEnemy(this.model.CurrentWorld, enemydata);
158  this.model.Hero = this.GetPlayer();
159  }
